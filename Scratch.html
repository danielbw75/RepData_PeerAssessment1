What is the average daily activity pattern?

#calculating the average of steps in each interval, by day
interval_group<-group_by(clndata, interval)
time_average<-summarize(interval_group, interval_mean = mean(steps))
time_average$interval<-formatC(time_average$interval/100,digits =2,
                               width =5, format="f", flag="0")

#making the plot
with(time_average, plot(interval, interval_mean,main= "Average of steps by interval of the day", xlab="time", ylab = "steps average", type = "l",col="blue"))

#finding wich time interval has the maximum average number of steps
subset(time_average, 
       time_average$interval_mean==max(time_average$interval_mean))[,"interval"]
